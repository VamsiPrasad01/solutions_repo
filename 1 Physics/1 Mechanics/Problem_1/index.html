def projectile_range(v0, theta_deg, g=g_earth): """Calculate range of projectile.""" theta_rad = np.radians(theta_deg) return (v0**2 * np.sin(2 * theta_rad)) / g

def trajectory(v0, theta_deg, g=g_earth, n_points=100): """Generate (x,y) trajectory coordinates.""" theta_rad = np.radians(theta_deg) t_flight = (2 * v0 * np.sin(theta_rad)) / g t = np.linspace(0, t_flight, n_points) x = v0 * np.cos(theta_rad) * t y = v0 * np.sin(theta_rad) * t - 0.5 * g * t**2 return x, y

def plot_3d_surface(): """Create 3D surface of range vs v0 and theta.""" fig = plt.figure(figsize=(14, 8)) ax = fig.add_subplot(111, projection='3d')
